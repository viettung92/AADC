function load_cml(){

   if [[ $# -ne 1 ]]; then
        echo "no valid name to swap to chosen; param num $#"
   else
        echo "Parameter #1 is $1"
        filepath="/home/audi_cup/Dokumente/aadc2018/src/aadcUser/"
        fileext="CMakeLists.txt"
        filename="$filepath"
        filename+="$1"
        filename+="_$fileext"
        usedfile="$filepath""$fileext"

        if [ -e "$filename" ]; then
            echo "Load file $filename to $usedfile"
            cp $filename $usedfile

        else
            echo "File $filename not found!"
        fi

   fi
}
